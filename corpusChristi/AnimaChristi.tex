\newHymn
\FirstLine{Anima Christi}
\settowidth{\versewidth}{Anima Christi, sanctifica me}

\begin{verse}[\versewidth]

\FirstVerse{A}{nima} Christi, sanctífica me.\\*
Corpus Christi, salva me.\\*
Sanguis Christi, inébria me.\\*
Aqua láteris Christi, lava me.
\pointtrans

\flagverse{2}Pássio Christi, confórta me.\\*
O bone Iesu, exáudi me.\\*
Intra tua vúlnera abscónde me.\\*
Ne permíttas me separári a te.

\flagverse{3}Ab hoste malígno defénde me.\\*
In hora mortis meae voca me.\\*
Et jube me veníre ad te,\\*
Ut cum Sanctis tuis laudem te\\*
In saécula saeculórum.\\*
Amen.

\end{verse}

\Hpoet{Ascribed to Pope John XXII}{1249--1334}

%Suffer me not to be separated from Thee
