\newHymn
\FirstLine{Stabat Mater dolorósa}

\begin{multicols}{2}
\begin{verse}[\versewidth]
%\raggedright
\FirstVerse{S}{tabat} Mater dolorósa\\*
Juxta crucem lacrimósa,\\*
Dum pendébat Fílius.
\pointtrans

\flagverse{2}Cujus ánimam geméntem,\\*
Contristátem et doléntem\\*
Pertransívit gládius.

\flagverse{3}O quam tristis et afflícta\\*
Fuit illa benedícta\\*
Mater Unigéniti!

\flagverse{4}Qu\ae\ m\ae rébat et dolébat,\\*
Pia Mater, dum vidébat\\*
Nati p\oe nas ínclyti.

\flagverse{5}Quis est homo qui non\\*
\qquad fleret,\\*
Matrem Christi si vidéret\\*
In tanto supplício?

\flagverse{6}Quis non posset contristári,\\*
Christi Matrem contemplári\\*
Doléntem cum Fílio?

\flagverse{7}Pro peccátis su\ae\ gentis,\\*
Vidit Jesum in torméntis,\\*
Et flagéllus súbditum.

\flagverse{8}Vidit suum dulcem Natum\\*
Moriéndo desolátum,\\*
Dum emísit spíritum.

\flagverse{9}Eia Mater, fons amóris,\\*
Me sentíre vim dolóris\\*
Fac, ut tecum lúgeam.

\flagverse{10}Fac ut árdeat cor meum\\*
In amándo Christum Deum,\\*
Ut sibi compláceam.

\flagverse{11}Sancta Mater, istud agas,\\*
Crucifíxi fige plagas\\*
Cordi meo válide.

\flagverse{12}Tui Nati vulneráti,\\*
Tam dignáti pro me pati,\\*
P\oe nas mecum dívide.

\flagverse{13}Fac me tecum pie flere,\\*
Crucifíxo condolére,\\*
Donec ego víxero.

\flagverse{14}Juxta Crucem tecum stare,\\*
Et me tibi sociáre\\*
In planctu desídero.

\flagverse{15}Virgo vírginem pr\ae clára,\\*
Mihi jam non sis amára:\\*
Fac me tecum plángere.

\flagverse{16}Fac ut portem Christi\\*
\qquad mortem\\*
Passiónis fac consórtem,\\*
Et plagas recólere.

\flagverse{17}Fac me plagis vulnerári,\\*
Fac me Cruc\emph{e} inebriári,\\*
Et cruóre Fílii.

\flagverse{18}Flammis ne urar succénsus,\\*
Per te, Virgo, sim defénsus\\*
In die judícii.

\flagverse{19}Christe, cum sit hinc exíre,\\*
Da per Matrem me veníre\\*
Ad palmam victóri\ae .

\flagverse{20}Quando corpus moriétur,\\*
Fac ut ánim\ae\ donétur\\*
Paradísi glória.  Amen.

\end{verse}
\end{multicols}

\Hpoet{Jacapone da Todi}{1230--1306}
