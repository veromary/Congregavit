\newHymn
\FirstLine{Ride on, ride on in majesty}

\begin{verse}[\versewidth]

\FirstVerse{R}{ide on!} ride on in majesty!\\
Hark! all the tribes hosanna cry;\\
O Savior meek, pursue thy road\\
with palms and scattered garments strowed.

\flagverse{2}Ride on! ride on in majesty!\\
In lowly pomp ride on to die;\\
O Christ, thy triumphs now begin\\
o'er captive death and conquered sin.

\flagverse{3}Ride on! ride on in majesty!\\
The angel-squadrons of the sky\\
look down with sad and wondering eyes\\
to see the approaching sacrifice.

\flagverse{4}Ride on! ride on in majesty!\\
Thy last and fiercest strife is nigh;\\
the Father on his sapphire throne\\
expects his own anointed Son.

\flagverse{5}Ride on! ride on in majesty!\\
In lowly pomp ride on to die;\\
bow thy meek head to mortal pain,\\
then take, O God, thy power, and reign.

\end{verse}

\Hpoet{Henry Hart Milman}{1791--1868}
