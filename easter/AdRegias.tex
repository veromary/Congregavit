\newHymn

\FirstLine{Ad regias Agni dapes}

\begin{multicols}{2}
%\settowidth{\versewidth}{His the scepter his the throne}
\begin{verse}[\versewidth]

\FirstVerse{A}{d} regias Agni dapes,\\*
Stolis amicti candidis\\
Post transitam maris Rubri\\*
Christo canamas Principi.\pointtrans

\flagverse{2}Divina cujus caritas\\*
Sacrum propinat\\> sanguinem,\\
Almique membra corporis\\*
Amor sacerdos immolat.

\flagverse{3}Sparsum cruorem postibus\\*
Vastator horret Angelus:\\
Fugitque divisum mare,\\*
Merguntur hostes\\> fluctibus.

\flagverse{4}Jam Pascha nostrum\\> Christus est,\\*
Paschalis idem victima:\\
Et pura puris mentibus\\*
Sinceritatis azyma.

\flagverse{5}O vera caeli victima,\\*
Subjecta cui sunt tartara,\\
Soluta mortis vincula,\\*
Recepta vitae praemia.

\flagverse{6}Victor subactis inferis\\*
Trophaea Christus\\> explicat,\\
Caeloque aperto, subditum\\
Regem tenebrarum trahit.

\flagverse{7}Ut sis perenne mentibus\\*
Paschale, Jesu, gaudium,\\
A morte dira criminum\\*
Vitae renatos libera.


\flagverse{8}Deo Patri sit gloria,\\*
Et Filio, qui a mortuis\\
Surrexit, ac Paraclito,\\*
In sempiterna saecula.

\end{verse}

\Hpoet{Ambrosian}{7th century}

\end{multicols}
