 \newHymn


\FirstLine{Hail, bright star of ocean}

\settowidth{\versewidth}{Of the Lord most high.}
\begin{multicols}{2}
\begin{verse}[\versewidth]

	\FirstVerse{H}{ail,} bright star of ocean,\\*
 God's own Mother blest,\\
Ever sinless Virgin,\\*
 Gate of heav'nly rest.
\pointorig

\flagverse{2}Taking that sweet Ave,\\*
    Which from Gabriel came,\\
        Peace confirm within us,\\*
    Changing Eva's name.
    
\flagverse{3}Break the captive's fetters:\\*
    Light on blindness pour;\\
        All our ills expelling,\\*
    Every bliss implore.
    
\flagverse{4}Show thyself a mother;\\*
    May the Word divine,\\
        Born for us thine Infant,\\*
    Hear our pray'rs thro' thine.
    
\flagverse{5}Virgins all excelling,\\*
    Mildest of the mild,\\
        Free from guilt preserve us,\\*
    Meek and undefiled.
    
\flagverse{6}Keep our life all spotless,\\*
    Make our way secure;\\
        Till we find in Jesus,\\*
    Joy for ever more.
    
\flagverse{7}Thro' the highest heaven,\\*
    To th'Almighty Three,\\
        Father, Son, and Spirit,\\*
    One same glory be.

\end{verse}
\end{multicols}

\Htrans{Edward Caswall}{1814--78}
