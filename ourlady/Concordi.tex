\FirstVerse{C}{oncórdi} lætítia,\\
Propúlsa mæstítia,\\
Maríæ præcónia\\
Récolat Ecclésia:\\
~ \quad Virgo María.

\flagverse{2}Quæ felíci gaúdio,\\
Resurgénte Dómino,\\
Floruit ut lílium:\\
Vivum cernens Fílium:\\
\ \quad Virgo María.

\end{verse}
\end{multicols}

\flagverse{3}Quam concéntu párili\\*
Chori láudant cóelici,\\
Et nos cum cæléstibus,\\*
Novum melos pángimus;\\*
\ \quad Virgo María.

\flagverse{4}O Regína Vírginum,\\*
Votis fave súpplicum,\\
Et post mortis stádium,\\*
Vitae cónfer práemium:\\*
\ \quad Virgo María.

\flagverse{5}Glóriosa Trínitas,\\*
Indivísa Unitas,\\
