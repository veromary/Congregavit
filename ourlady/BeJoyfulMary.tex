
\newHymn

\FirstLine{Be joyful, Mary, heav’nly Queen}

\begin{verse}[\versewidth]

\FirstVerse{B}{e joyful,} Mary, heav’nly Queen,\\
\vin Gaude, Maria!\\
Now let thine eyes with gladness beam,\\
Alleluia! Lætare, O Maria!

\flagverse{2}Whom thou didst bear by heaven’s grace,\\
\vin Gaude, Maria!\\
Did all our guilt and sin efface,\\
Alleluia! Lætare, O Maria!

\flagverse{3}The Lord has risen from the dead,\\
\vin Gaude, Maria!\\
He rose with might as He had said,\\
Alleluia! Lætare, O Maria!

\flagverse{4}O pray to God, thou Virgin fair,\\
\vin Gaude, Maria!\\
That He our souls to heaven bear,\\
Alleluia! Lætare, O Maria!

\end{verse}
